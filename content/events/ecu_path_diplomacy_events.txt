# Diplo gatekeeper
planet_event = {
	id = ef_diplo.10
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_list = {
			24 = {} # 1 in 25 of something happening - we don't have that many event so we have to ration them
			1 = {
				random_list = {
					1 = { planet_event = { id = ef_diplo.20 } }
					1 = { planet_event = { id = ef_diplo.30 } }
					1 = { planet_event = { id = ef_diplo.50 } }
					1 = { planet_event = { id = ef_diplo.70 }}
				}
			}
		}
		# Roll for movies
		if = {
			limit = {
				has_planet_flag = ef_space_hollywood
			}
			random_list = {
				14 = {  }
				# 1 in 15 chance of a movie - there's also triggers and cooldowns so will be less than this in practice
				1 = {
					random_list = {
						1 = { planet_event = { id = ef_diplo.51 }}
						1 = { planet_event = { id = ef_diplo.52 }}
						1 = { planet_event = { id = ef_diplo.53 }}
						1 = { planet_event = { id = ef_diplo.54 }}
					}
				}
			}
		}
		# Refresh
		planet_event = { id = ef_diplo.10 days = 360 }
	}
}

# Playground of the elite
planet_event = {
	id = ef_diplo.20
	title = ef_diplo.20.name
	desc = "ef_diplo.20.desc"
	picture  = GFX_evt_crisis_defeated
	location = root
	is_triggered_only = yes

	trigger = {
		not = { has_planet_flag = ef_recent_playground_of_the_elite}
	}

	immediate = {
		set_timed_planet_flag = { flag = ef_recent_playground_of_the_elite years = 25 }
	}

	# Acknowledge
	option = {
		name = ef_diplo.20.a
		add_modifier = { modifier = ef_economic_boom years = 5 }
	}

	# Emphasise our status
	option = {
		name = ef_diplo.20.b
		add_modifier = { modifier = ef_playground_of_the_elite }
        allow = {
			owner = {
				resource_stockpile_compare = { resource = influence value >= 150 }
			}
		}
		owner = {
			add_resource = { influence = -150 }
		}
	}
}

# Badly written Khan expy
planet_event = {
	id = ef_diplo.53
	title = ef_diplo.53.name
	desc = "ef_diplo.53.desc"
	picture  = GFX_evt_khan_throne_room
	location = root
	is_triggered_only = yes

	trigger = {
		has_planet_flag = ef_space_hollywood
		NOT = {
			owner = { has_country_flag = ef_recent_khan_expy}
		}
		any_country = {
			OR = {
					is_country_type = awakened_marauders
					is_country_type = dormant_marauders
					is_country_type = ruined_marauders
					is_country_type = marauder_raiders
			}
		}
	}

	immediate = {
		random_country = {
			limit = {
				OR = {
					is_country_type = awakened_marauders
					is_country_type = dormant_marauders
					is_country_type = ruined_marauders
					is_country_type = marauder_raiders
				}
			}
			save_event_target_as = ef_marauder_expy_country
		}
		owner = {
			set_timed_country_flag = {
				flag = ef_recent_khan_expy
				years = 50 # We don't want this one to pop too often
			}
		}
	}

	# Acknowledge
	option = {
		name = ef_diplo.53.a
		owner = {
			add_resource = {
				unity = 5000
			}
		}
	}
}

# We've attracted cultural talent
planet_event = {
	id = ef_diplo.70
	title = ef_diplo.70.name
	desc = "ef_diplo.70.desc"
	picture  = GFX_evt_announcement
	location = root
	is_triggered_only = yes

	trigger = {
		NOT = {
			owner = { has_country_flag = ef_recent_cultural_talent}
		}
	}

	immediate = {
		owner = {
			set_timed_country_flag = {
				flag = ef_recent_cultural_talent
				years = 25
			}
		}
		random_galaxy_species = {
			save_event_target_as = ef_talented_creative_species
			weights = {
				base = 100
				modifier = {
					mult = 0
					# We don't want random aliens we've never seen before showing up.
					NOT = {
						any_country = {
							is_same_species = prev
							has_established_contact = root.owner
						}
					}
				}
				# Fallback only
				modifier = {
					mult = 0.01
					is_same_species = root.owner
				}
			}
		}
	}

	# Wow! Sponsor them.
	option = {
		name = ef_diplo.70.a
		custom_tooltip = ef_diplo.70.a.tt
		owner = {
			create_leader = {
				class = official
				species = event_target:ef_talented_creative_species
				name = random
				skill = 6
				background_job = entertainer
				background_ethic = ethic_xenophile
				randomize_traits = no
				traits = {
					trait = subclass_official_delegate
					trait = leader_trait_ef_artist_2
				}
			}
		}
	}

	# Let's get a local talent instead
	option = {
		name = ef_diplo.70.b
		custom_tooltip = ef_diplo.70.b.tt
		trigger = {
			owner = {
				is_xenophobe = yes
			}
		}
		owner = {
			create_leader = {
				class = official
				species = owner_main_species
				name = random
				skill = 4
				background_job = entertainer
				background_ethic = ethic_xenophile
				randomize_traits = no
				traits = {
					trait = subclass_official_delegate
					trait = leader_trait_ef_artist_1
				}
			}
		}
	}

	# We're not gonna sponsor them, just enjoy the benefits
	option = {
		name = ef_diplo.70.c
		add_modifier = { modifier = ef_thriving_cultural_scene years = 10 }
	}
}
