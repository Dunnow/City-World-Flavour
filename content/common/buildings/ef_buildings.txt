building_ef_hologem_bourse = {
	base_buildtime = 360
	exempt_from_ai_planet_specialization = no
	category = trade
	base_cap_amount = 1

	building_sets = {
		trade
	}

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
		NOT = { has_building = building_low_tech_capital }
		has_planet_flag = ef_has_hologems_industry
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
            NOT = { has_planet_flag = ef_has_hologems_industry }
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		job_clerk_add = 100
		job_ef_gemcutter_add = 200
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		text = job_ef_gemcutter_effect_desc
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}
}
