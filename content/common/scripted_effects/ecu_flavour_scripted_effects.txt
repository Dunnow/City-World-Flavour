            limit = {
                not = {  has_deposit = d_ef_orbital_control_network  }
            }
            add_modifier = { modifier = ef_crowded_spacelanes days = -1 }
        }
        if = {
            limit = {
                not = {  has_deposit = d_ef_climate_management_centre  }
            }
            add_modifier = { modifier = ef_unstable_climate days = -1 }
        }
        if = {
            limit = {
                not = {  has_deposit = d_ef_seismic_stabilization_stations  }
            }
            add_modifier = { modifier = ef_vertical_sprawl days = -1 }
        }
    }
}

# Planet Scope
ef_set_deep_levels_modifier= {
    remove_modifier = ef_deep_levels_1
    remove_modifier = ef_deep_levels_2
    remove_modifier = ef_deep_levels_3
    remove_modifier = ef_deep_levels_4
    remove_modifier = ef_deep_levels_5
    
    if = {
        limit = {
            check_variable_arithmetic = {
                which = ef_deep_level_count
                divide = trigger:planet_size
                value > 0
            }
        }
        # Minor Deep Levels
        if = {
            limit = {
                AND ={
                    check_variable_arithmetic = {
                        which = ef_deep_level_count
                        divide = trigger:planet_size
                        value < 0.25
                    }
                }
            }
            add_modifier = { modifier = ef_deep_levels_1 days = -1 }
        }
        # Moderate Deep Levels
        else_if = {
            limit = {
                check_variable_arithmetic = {
                    which = ef_deep_level_count
                    divide = trigger:planet_size
                    value < 0.5
                }
            }
            add_modifier = { modifier = ef_deep_levels_2 days = -1 }
        }
        # Major Deep Levels
        else_if = {
            limit = {
                check_variable_arithmetic = {
                    which = ef_deep_level_count
                    divide = trigger:planet_size
                    value < 0.75
                }
            }
            add_modifier = { modifier = ef_deep_levels_3 days = -1 }
        }
        # Enormous Deep Levels
        else_if = {
            limit = {
                check_variable_arithmetic = {
                    which = ef_deep_level_count
                    divide = trigger:planet_size
                    value < 1
                }
            }
            add_modifier = { modifier = ef_deep_levels_4 days = -1 }

        }
        # Sprawling
        else = {
            add_modifier = { modifier = ef_deep_levels_5 days = -1 }
        }
    }
}

# Works for relic or normal worlds. Has its own event announcement to state that this was natural growth
ef_grow_ecu = {
    # Make city-planet
    save_deposits = yes
    clear_deposits = yes
    if = {
        limit = {
            is_planet_class = pc_relic
        }
        add_deposit = d_former_relic_world
    }
    restore_saved_deposits = yes
    set_industrial_focus_flags = yes
    change_pc = pc_city
    relic_world_conversion_effect = {
        DISTRICT = district_arcology_housing
        VARIABLE = num_housing_districts
    }
    relic_world_conversion_effect = {
        DISTRICT = district_arcology_arms_industry
        VARIABLE = num_industrial_districts
    }
    set_ecu_industrial_districts_effect = yes
    set_planet_entity = {
        entity = "city_planet_entity"
        graphical_culture = owner
    }
    planet_event = { id = action.68 }
    planet_event = { id = ef_misc.6 }
    owner = {
        country_event = { id = distar.201 } # port for living planet distar.197, triggers in event
    }
    while = {
        count = 4
        solar_system = {
            random_system_ambient_object = {
                limit = { has_ambient_object_flag = relic_world_debris }
                destroy_ambient_object = this
            }
        }
    }
    if = {
        limit = {
            OR = {
                is_active_resolution = "resolution_ecology_environmental_control_board"
                is_active_resolution = "resolution_ecology_paradise_initiative"
            }
        }
        owner = {
            set_timed_country_flag = { flag = resolution_breached_terraformed days = 3600 }
        }
    }
    set_planet_flag = ef_naturally_grown_ecu
}
