		modifier = {
			add = -1

			current_situation_approach = ef_situation_normal_operations

		modifier = {
			add = -2
			desc = ef_situation_severe_response_tt

			25 = ef_trade.54
 		}
 	}
	abort_trigger = {
		OR = {
			NOT = { exists = target.owner }
			target.owner = { NOT = { is_same_value = root.owner } }
		}
	}
